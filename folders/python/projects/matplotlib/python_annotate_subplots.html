# Get the max and min values in this dataset
max_date = monthly_df.loc[monthly_df['close'] == monthly_df['close'].max()].index.values
min_date = monthly_df.loc[monthly_df['close'] == monthly_df['close'].min()].index.values
In [7]:
# Get the actual max and min close values
max_close = round(monthly_df['close'].max(),2)
min_close = round(monthly_df['close'].min(),2)
In [8]:
fig = plt.figure(figsize=(10,10))
ax = fig.add_subplot(1,1,1)

# Create the plot at the axes instance with a line style - and line color blue
monthly_df.plot(ax=ax, style='b-')

metric_data = [(max_date, f'Max Close ${max_close}'), (min_date, f"Min Close ${min_close}")]

for date, label in metric_data:
    ax.annotate(label, xy=(date[0], monthly_df.asof(date[0])),
                xytext=(date[0], monthly_df.asof(date[0]) + 3),
                arrowprops=dict(facecolor='black', headwidth=10, width=2, headlength=4),
                horizontalalignment='right', verticalalignment='bottom')

plt.legend(loc='upper left')    
plt.show()

How does this work?!¶

fig = plt.figure(figsize=(10,10))

  • This code will create a figure object with a set figsize
    ax = fig.add_subplot(1,1,1)
  • This will create an instance of a single axes object with 1 row, 1 col, and at index 1
    monthly_df.plot(ax=ax, style='b-')
  • creates a line plot at the axes instance that is blue 'b' and has a solid linestyle
    metric_data = [(max_date, "Max Close"), (min_date, "Min Close")]
  • This is a list of tuples that contain the date of the max close and min close with the respective value
    for date, label in metric_data:
  • this is the for loop that will go through the tuples in the metric_data list
    ax.annotate(label, xy=(date[0], monthly_df.asof(date[0])),
  • This code adds the annotation and uses the tuple data that was pulled out of the metric_data list
  • the label is the string
  • xy takes in a coordinate of the date
    xytext=(date[0], monthly_df.asof(date[0]) + 3),
  • this code will place the actual text along an x and y coordinate. I'm adding 3 in the y axis so there is room for the arrow annotation
    arrowprops=dict(facecolor='black', headwidth=8, width=2, headlength=4), horizontalalignment='right', verticalalignment='bottom')
  • This code will add the text arrow annotation
  • documentation for each argument: https://matplotlib.org/3.1.1/api/_as_gen/matplotlib.pyplot.annotate.html
