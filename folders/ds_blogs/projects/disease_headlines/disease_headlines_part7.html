# Function takes in headlines and then returns the count of headlines with Zika
def count_zika_mentions(headlines):
#   Regular expression will match word "Zika" in a headline, and will be be case-insensitive.
    zika_regex = re.compile(r'\bzika\b',
                            flags=re.IGNORECASE)
    zika_count = 0
    for headline in headlines:
        if zika_regex.search(headline):
            zika_count += 1

    return zika_count
